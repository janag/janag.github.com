Load the correct server module depending upon HTTPS
             		                            
primaryServer = ->
  
  if SS.config.https.enabled
    ssl = require('./ssl')
    out =  
      server:    middleware.primary(ssl.keys.options())
      config:    SS.config.https
      protocol:  'https'
  else
    out =
      server:    middleware.primary()
      config:    SS.config.http
      protocol:  'http'
  
  out
